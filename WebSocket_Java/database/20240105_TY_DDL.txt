DROP TABLE IF EXISTS ST_SURVEILLANCE_VIDEO;
CREATE TABLE ST_SURVEILLANCE_VIDEO(
    `ID` INT NOT NULL AUTO_INCREMENT  COMMENT '主键' ,
    `VIDEO_NAME` VARCHAR(255)    COMMENT '视频站点名称' ,
    `BRAND` VARCHAR(255)    COMMENT '品牌' ,
    `MODEL_NUMBER` VARCHAR(255)    COMMENT '型号' ,
    `PORT` INT    COMMENT '端口号' ,
    `CIDEOUSERNAME` VARCHAR(32)    COMMENT '用户名' ,
    `VIDEOPASSWORD` VARCHAR(255)    COMMENT '密码' ,
    `IP` VARCHAR(255)    COMMENT 'IP' ,
    `ELECTRICITY_METER` VARCHAR(255)    COMMENT '电表卡号' ,
    `STLC` VARCHAR(900)    COMMENT '站点位置' ,
    `RIVERSTREAM` VARCHAR(255)    COMMENT '所在水系及河' ,
    `ORDERID` INT    COMMENT '排序' ,
    `CREATED_BY` VARCHAR(32)    COMMENT '创建人' ,
    `CREATED_TIME` DATETIME    COMMENT '创建时间' ,
    `UPDATED_BY` VARCHAR(32)    COMMENT '更新人' ,
    `UPDATED_TIME` DATETIME    COMMENT '更新时间' ,
    PRIMARY KEY (ID)
)  COMMENT = '监控视频站点表';
